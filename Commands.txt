"svn" + username			= Client to server; new user with username has connected
"cnu" + name				= Server sent message: new client of "name" has connected (Client writes connection message)
"svt"					= Client to server; client has left || Server to client; server shut down
"cul" + number				= Server to client; Client(number) has left; remove them from member list
"msg" + sender number + "n" + message	= server to client; message to general chat
"pmg" + sender number + "n" + message	= server to client; pm to this sender
